import { connectDB } from './lib/mongodb';
import { UserModel } from './models';
import MenuItemModel from './models/menu-item-model';
import InventoryModel from './models/inventory-model';
await connectDB();
const userCount = await UserModel.countDocuments();
const menuCount = await MenuItemModel.countDocuments();
const inventoryCount = await InventoryModel.countDocuments();
console.log('📊 Database Status:');
console.log('Users:', userCount);
console.log('Menu Items:', menuCount, '(Expected: 62 - 46 drinks + 16 food)');
console.log('Inventory Records:', inventoryCount, '(Expected: 62)');
process.exit(0);
"
